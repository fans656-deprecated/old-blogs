          一个礼拜前曾许愿把renju（五子棋）编出来。终于实现。先哈哈个把分钟。
          第一次，用了一上午时间。虽然简陋到不行，但仍是好歹出来了。我不嫌丑，不嫌丑~~~~`````
          hoho,而且，这个不怕G小姐批评指正了。
 
 
        
 
          文本模式，C40。无AI。
          模块化...
          棋盘用二维数组，定义了一大堆Hex常量。
          用init()函数初始化，getkey()获得按键值，move()移动光标，down()落子，judgewin()判断胜利，gameover()结束游戏。
          哇，这么写觉得真够简单的。要编时也如此多快活。
         
 
 
 
 
 
#include<stdio.h>
#include<conio.h>
#include<bios.h>
#include<process.h>
#include<stdlib.h>
 
#define LT 218
#define LB 192
#define RT 191
#define RB 217
#define LL 195
#define RL 180
#define TL 194
#define BL 193
#define CR 197
#define BLACKUP     0x4800
#define BLACKDOWN   0x5000
#define BLACKLEFT   0x4B00
#define BLACKRIGHT  0x4D00
#define BLACKENTER  0x1c0d
#define WHITEUP     0x1177
#define WHITEDOWN   0x1f73
#define WHITELEFT   0x1e61
#define WHITERIGHT  0x2064
#define WHITEENTER  0x3920
#define ESC        0x11b
#define BLACK 1
#define WHITE 2
 
void init();
int getkey();
void move();
void down(char board[][15]);
void gameover();
void judgewin();
 
char board[15][15];
char order=BLACK;
 
int main()
{
 init();
 while(1)
 {
  move();
 }
}
void init()
{
 int row,colum;
 board[0][0]=LT;board[14][0]=LB;board[0][14]=RT;board[14][14]=RB;
 for(row=0,colum=1;colum<14;colum++)
  board[row][colum]=TL;
 for(row=14,colum=1;colum<14;colum++)
  board[row][colum]=BL;
 for(colum=0,row=1;row<14;row++)
  board[row][colum]=LL;
 for(colum=14,row=1;row<14;row++)
  board[row][colum]=RL;
 for(row=1;row<14;row++)
  for(colum=1;colum<14;colum++)
   board[row][colum]=CR;
 clrscr();
 textmode(C40);
 _setcursortype(_SOLIDCURSOR);
 for(row=0;row<15;row++)
 {
  for(colum=0;colum<15;colum++)
   putchar(board[row][colum]);
  putchar('\n');
 }
 gotoxy(20,8);
 puts("MADE BY fans656");
 gotoxy(1,17);
 puts("P1          P2");
 puts("UP          W");
 puts("DOWN        S          Quit:Esc");
 puts("LEFT        A");
 puts("RIGHT       D");
 puts("Enter      Space");
 gotoxy(8,8);
}
int getkey()
{
 return(bioskey(0));
}
void move()
{
 int x,y;
 x=wherex();y=wherey();
 switch(getkey())
 {
  case BLACKUP:
   if(order==BLACK)
    if(y!=1)gotoxy(x,--y);
   break;
  case BLACKDOWN:
   if(order==BLACK)
    if(y!=15)gotoxy(x,++y);
   break;
  case BLACKLEFT:
   if(order==BLACK) 
    if(x!=1)gotoxy(--x,y);
   break;
  case BLACKRIGHT:
   if(order==BLACK)
    if(x!=15)gotoxy(++x,y);
   break;
  case BLACKENTER:
   if(order==BLACK)
    down(board);
   break;
  case WHITEUP:
   if(order==WHITE)
    if(y!=1)gotoxy(x,--y);
   break;
  case WHITEDOWN:
   if(order==WHITE)
    if(y!=15)gotoxy(x,++y);
   break;
  case WHITELEFT:
   if(order==WHITE) 
    if(x!=1)gotoxy(--x,y);
   break;
  case WHITERIGHT:
   if(order==WHITE)
    if(x!=15)gotoxy(++x,y);
   break;
  case WHITEENTER:
   if(order==WHITE)
    down(board);
   break;
  case ESC:
   order=' ';gameover();break;
  default:;
 }
}
void down(char board [][15])
{
 int row,colum;
 row=wherey()-1;colum=wherex()-1;
 if(board[row][colum]!=BLACK&&board[row][colum]!=WHITE)
 {
  if(order==BLACK){putchar(board[row][colum]=BLACK);order=WHITE;}
  else   {putchar(board[row][colum]=WHITE);order=BLACK;} 
  gotoxy(colum+1,row+1);
 }
 judgewin();
}
void judgewin()
{
 int  i,j,m,n;
 char current;
 i=wherey()-1;j=wherex()-1;
 current=board[i][j]; 
 for(n=0;board[i][--j]==current;n++);
 i=wherey()-1;j=wherex()-1;
 for(m=0;board[i][++j]==current;m++);
 if(m+n==4)gameover();
 i=wherey()-1;j=wherex()-1;
 for(n=0;board[--i][j]==current;n++);
 i=wherey()-1;j=wherex()-1;
 for(m=0;board[++i][j]==current;m++);
 if(m+n==4)gameover();
 i=wherey()-1;j=wherex()-1;
 for(n=0;board[--i][--j]==current;n++);
 i=wherey()-1;j=wherex()-1;
 for(m=0;board[++i][++j]==current;m++);
 if(m+n==4)gameover();
 i=wherey()-1;j=wherex()-1; 
 for(n=0;board[--i][++j]==current;n++);
 i=wherey()-1;j=wherex()-1;
 for(m=0;board[++i][--j]==current;m++);
 if(m+n==4)gameover();
 i=wherey()-1;j=wherex()-1;
   
}
void gameover()
{
 _setcursortype(_NOCURSOR);
 if(order==WHITE) {gotoxy(22,4);printf("Black Win!");getch();}
 else if(order==BLACK) {gotoxy(22,4);printf("White Win!");getch();}
 exit(0);
}
